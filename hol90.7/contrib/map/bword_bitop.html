Definitions

WNOT_DEF 
|- !l. WNOT (WORD l) = WORD (MAP ~ l)
WAND_DEF 
|- !l1 l2. WORD l1 WAND WORD l2 = WORD (MAP2 $/\ l1 l2)
WOR_DEF 
|- !l1 l2. WORD l1 WOR WORD l2 = WORD (MAP2 $\/ l1 l2)
WXOR_DEF 
|- !l1 l2. WORD l1 WXOR WORD l2 = WORD (MAP2 (\x y. ~(x = y)) l1 l2)
