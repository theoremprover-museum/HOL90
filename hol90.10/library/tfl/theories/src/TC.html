Definitions

TC_DEF 
|- !R a b.
     TC R a b =
     (!P.
       (!x y. R x y ==> P x y) /\ (!x y z. P x y /\ P y z ==> P x z) ==>
       P a b)
transitive_def 
|- !R. transitive R = (!x y z. R x y /\ R y z ==> R x z)
