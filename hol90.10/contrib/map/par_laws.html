PREFIX_PAR_1 
|- !c P Q.
     c IN ALPHA P INTER ALPHA Q ==>
     ((c --> P) PAR (c --> Q) = c --> (P PAR Q))
PREFIX_PAR_2 
|- !c d P Q.
     {c; d} SUBSET ALPHA P INTER ALPHA Q /\ ~(c = d) ==>
     ((c --> P) PAR (d --> Q) = STOP (ALPHA P UNION ALPHA Q))
